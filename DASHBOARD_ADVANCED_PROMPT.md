```php
// app/Http/Controllers/DashboardController.php
<?php

namespace App\Http\Controllers;

use App\Services\DecisionSupport\EnhancedRecommendationEngine;
use App\Services\Psychometric\RiasecAnalyzer;
use App\Services\Explainability\NarrativeGenerator;
use App\Services\Analytics\SensitivityAnalyzer;
use App\Services\Comparison\AlgorithmComparator;
use App\Services\Evidence\EmpiricalValidator;
use Illuminate\Http\Request;
use Inertia\Inertia;

class DashboardController extends Controller
{
    public function __construct(
        private EnhancedRecommendationEngine $recommendationEngine,
        private RiasecAnalyzer $riasecAnalyzer,
        private NarrativeGenerator $narrativeGenerator,
        private SensitivityAnalyzer $sensitivityAnalyzer,
        private AlgorithmComparator $algorithmComparator,
        private EmpiricalValidator $empiricalValidator
    ) {}
    
    public function show(int $assessmentId)
    {
        // Fetch assessment with all relationships
        $assessment = Assessment::with([
            'user',
            'recommendationResults.major',
            'psychometricProfile',
            'ahpWeights',
            'topsisCalculation'
        ])->findOrFail($assessmentId);
        
        // SECTION 1: Executive Summary
        $executiveSummary = $this->buildExecutiveSummary($assessment);
        
        // SECTION 2: Explainable AI Narrative
        $aiNarrative = $this->narrativeGenerator->generate(
            $assessment->psychometricProfile,
            $assessment->recommendationResults->first(),
            $assessment->ahpWeights,
            $assessment->criteriaScores
        );
        
        // SECTION 3: Top 10 Recommendations with Enhanced Metrics
        $recommendations = $this->enrichRecommendations(
            $assessment->recommendationResults->take(10)
        );
        
        // SECTION 4: Multi-Dimensional Visualizations
        $visualizations = $this->buildVisualizationData($assessment);
        
        // SECTION 5: Psychometric Deep Dive
        $psychometricAnalysis = $this->riasecAnalyzer->generateDeepDive(
            $assessment->psychometricProfile
        );
        
        // SECTION 6: Sensitivity Analysis
        $sensitivityData = $this->sensitivityAnalyzer->performAnalysis(
            $assessment->psychometricProfile,
            $assessment->ahpWeights
        );
        
        // SECTION 7: Algorithm Transparency
        $algorithmDetails = $this->buildAlgorithmTransparency($assessment);
        
        // SECTION 8: Comparative Algorithm Analysis
        $algorithmComparison = $this->algorithmComparator->compareAll(
            $assessment->psychometricProfile,
            $assessment->ahpWeights
        );
        
        // SECTION 9: Evidence-Based Justification
        $empiricalEvidence = $this->empiricalValidator->gatherEvidence(
            $assessment->recommendationResults->first()->major,
            $assessment->psychometricProfile
        );
        
        // SECTION 10: Actionable Next Steps
        $actionPlan = $this->generateActionPlan($assessment);
        
        return Inertia::render('Dashboard/AdvancedDashboard', [
            'assessment' => $assessment,
            'executiveSummary' => $executiveSummary,
            'aiNarrative' => $aiNarrative,
            'recommendations' => $recommendations,
            'visualizations' => $visualizations,
            'psychometricAnalysis' => $psychometricAnalysis,
            'sensitivityData' => $sensitivityData,
            'algorithmDetails' => $algorithmDetails,
            'algorithmComparison' => $algorithmComparison,
            'empiricalEvidence' => $empiricalEvidence,
            'actionPlan' => $actionPlan,
            'performanceMetrics' => $this->getPerformanceMetrics()
        ]);
    }
    
    private function buildExecutiveSummary(Assessment $assessment): array
    {
        $topRecommendation = $assessment->recommendationResults->first();
        
        return [
            'major_name' => $topRecommendation->major->name,
            'major_code' => $topRecommendation->major->code,
            'match_score_percentage' => round($topRecommendation->final_score * 100, 1),
            'closeness_coefficient' => round($topRecommendation->topsis_score, 3),
            'ml_confidence' => round($topRecommendation->ml_success_probability * 100, 1),
            'consistency_ratio' => round($assessment->ahpWeights->consistency_ratio, 3),
            'consistency_status' => $this->getConsistencyStatus($assessment->ahpWeights->consistency_ratio),
            'rank' => 1,
            'total_alternatives' => $assessment->recommendationResults->count()
        ];
    }
    
    private function enrichRecommendations($recommendations)
    {
        return $recommendations->map(function ($rec) {
            return [
                'rank' => $rec->rank,
                'major_id' => $rec->major_id,
                'major_name' => $rec->major->name,
                'major_description' => $rec->major->description,
                'match_score' => round($rec->final_score * 100, 1),
                'topsis_score' => round($rec->topsis_score, 3),
                'profile_matching_score' => round($rec->profile_matching_score, 1),
                'ml_success_probability' => round($rec->ml_success_probability * 100, 1),
                'gap_analysis' => $rec->gap_analysis,
                'gap_total' => $rec->gap_analysis['total_gap'] ?? 0,
                'gap_status' => $this->getGapStatus($rec->gap_analysis['total_gap'] ?? 0),
                'ideal_positive_distance' => round($rec->ideal_positive_distance, 3),
                'ideal_negative_distance' => round($rec->ideal_negative_distance, 3),
                'mahalanobis_distance' => round($rec->mahalanobis_distance, 3),
                'behavioral_multiplier' => round($rec->behavioral_multiplier, 3),
                'criteria_breakdown' => $rec->criteria_breakdown,
                'strengths' => $this->identifyStrengths($rec),
                'weaknesses' => $this->identifyWeaknesses($rec),
                'compensations' => $this->identifyCompensations($rec)
            ];
        });
    }
    
    private function buildVisualizationData(Assessment $assessment): array
    {
        $topRecommendation = $assessment->recommendationResults->first();
        
        return [
            'radar_chart' => $this->buildRadarChartData($assessment, $topRecommendation),
            'waterfall_chart' => $this->buildWaterfallChartData($assessment, $topRecommendation),

            'decision_tree_visualization' => $this->buildDecisionTree($assessment)
        ];
    }
    
    private function buildRadarChartData(Assessment $assessment, $recommendation): array
    {
        $profile = $assessment->psychometricProfile;
        $major = $recommendation->major;
        
        return [
            'labels' => [
                'Kesiapan Akademik',
                'Kemampuan Logika',
                'Kemampuan Analitis',
                'Minat Bidang',
                'Konsistensi',
                'Toleransi Biaya'
            ],
            'datasets' => [
                [
                    'label' => 'Profil Anda',
                    'data' => [
                        $profile->academic_readiness,
                        $profile->logic_ability,
                        $profile->analytical_ability,
                        $profile->field_interest,
                        $profile->consistency,
                        $profile->cost_tolerance
                    ],
                    'backgroundColor' => 'rgba(59, 130, 246, 0.2)',
                    'borderColor' => 'rgb(59, 130, 246)',
                    'pointBackgroundColor' => 'rgb(59, 130, 246)'
                ],
                [
                    'label' => $major->name . ' (Threshold)',
                    'data' => [
                        $major->academic_threshold,
                        $major->logic_requirement,
                        $major->analytical_requirement,
                        $major->interest_alignment,
                        $major->consistency_requirement,
                        $major->cost_level
                    ],
                    'backgroundColor' => 'rgba(16, 185, 129, 0.2)',
                    'borderColor' => 'rgb(16, 185, 129)',
                    'pointBackgroundColor' => 'rgb(16, 185, 129)'
                ]
            ]
        ];
    }
    
    private function buildWaterfallChartData(Assessment $assessment, $recommendation): array
    {
        $criteriaContributions = $this->calculateCriteriaContributions(
            $assessment,
            $recommendation
        );
        
        $data = [
            ['label' => 'Base Score', 'value' => 50, 'type' => 'base']
        ];
        
        $runningTotal = 50;
        foreach ($criteriaContributions as $criterion => $contribution) {
            $data[] = [
                'label' => $criterion,
                'value' => $contribution,
                'type' => $contribution >= 0 ? 'positive' : 'negative',
                'running_total' => $runningTotal + $contribution
            ];
            $runningTotal += $contribution;
        }
        
        $data[] = [
            'label' => 'Final Score',
            'value' => $runningTotal,
            'type' => 'total'
        ];
        
        return $data;
    }
    
    private function buildDecisionTree(Assessment $assessment): array
    {
        $weights = $assessment->ahpWeights;
        $profile = $assessment->psychometricProfile;
        $topRec = $assessment->recommendationResults->first();
        
        // Find dominant criterion
        $dominantCriterion = collect($weights->weights)->sortDesc()->keys()->first();
        
        return [
            'root' => [
                'question' => "Prioritas utama: {$dominantCriterion}?",
                'user_answer' => "Ya (bobot {$weights->weights[$dominantCriterion]}%)",
                'weight' => $weights->weights[$dominantCriterion],
                'branches' => [
                    [
                        'question' => $this->getSecondaryQuestion($profile),
                        'user_answer' => $this->getSecondaryAnswer($profile),
                        'branches' => [
                            [
                                'question' => $this->getTertiaryQuestion($profile),
                                'user_answer' => $this->getTertiaryAnswer($profile),
                                'result' => [
                                    'major' => $topRec->major->name,
                                    'score' => round($topRec->final_score * 100, 1) . '%',
                                    'confidence' => 'High'
                                ]
                            ]
                        ]
                    ]
                ]
            ]
        ];
    }
    
    private function getConsistencyStatus(float $cr): array
    {
        if ($cr < 0.05) {
            return [
                'label' => 'Excellent',
                'color' => 'green',
                'icon' => '✅',
                'description' => 'Preferensi Anda sangat konsisten dan logis'
            ];
        } elseif ($cr < 0.08) {
            return [
                'label' => 'Very Good',
                'color' => 'blue',
                'icon' => '✓',
                'description' => 'Preferensi Anda konsisten'
            ];
        } elseif ($cr < 0.10) {
            return [
                'label' => 'Good',
                'color' => 'yellow',
                'icon' => '○',
                'description' => 'Preferensi Anda cukup konsisten'
            ];
        } else {
            return [
                'label' => 'Needs Review',
                'color' => 'red',
                'icon' => '⚠',
                'description' => 'Preferensi Anda perlu dikalibrasi ulang'
            ];
        }
    }
    
    private function getGapStatus(float $gap): array
    {
        if ($gap >= 0) {
            return [
                'label' => 'Exceed',
                'color' => 'green',
                'icon' => '↑',
                'description' => 'Anda melebihi persyaratan'
            ];
        } elseif ($gap >= -10) {
            return [
                'label' => 'Minor Deficit',
                'color' => 'blue',
                'icon' => '→',
                'description' => 'Sedikit di bawah persyaratan'
            ];
        } elseif ($gap >= -20) {
            return [
                'label' => 'Moderate Deficit',
                'color' => 'yellow',
                'icon' => '↓',
                'description' => 'Cukup di bawah persyaratan'
            ];
        } else {
            return [
                'label' => 'Significant Deficit',
                'color' => 'red',
                'icon' => '↓↓',
                'description' => 'Jauh di bawah persyaratan'
            ];
        }
    }
}
```

### Service Layer: Narrative Generator

```php
// app/Services/Explainability/NarrativeGenerator.php
<?php

namespace App\Services\Explainability;

use App\Models\PsychometricProfile;
use App\Models\RecommendationResult;

class NarrativeGenerator
{
    public function generate(
        PsychometricProfile $profile,
        RecommendationResult $recommendation,
        $ahpWeights,
        $criteriaScores
    ): array {
        return [
            'executive_summary' => $this->generateExecutiveSummary($recommendation),
            'why_this_major' => $this->generateWhyThisMajor($profile, $recommendation, $ahpWeights),
            'strength_analysis' => $this->generateStrengthAnalysis($profile, $recommendation),
            'weakness_compensation' => $this->generateWeaknessCompensation($profile, $recommendation),
            'riasec_narrative' => $this->generateRiasecNarrative($profile->riasec_profile),
            'success_prediction' => $this->generateSuccessPrediction($recommendation),
            'career_outlook' => $this->generateCareerOutlook($recommendation->major),
            'learning_environment' => $this->generateLearningEnvironment($recommendation->major),
            'challenges_opportunities' => $this->generateChallengesOpportunities($profile, $recommendation)
        ];
    }
    
    private function generateExecutiveSummary(RecommendationResult $recommendation): string
    {
        $major = $recommendation->major->name;
        $score = round($recommendation->final_score * 100, 1);
        $confidence = round($recommendation->ml_success_probability * 100, 1);
        
        return "MajorMind merekomendasikan **{$major}** sebagai pilihan terbaik Anda dengan " .
               "tingkat kesesuaian **{$score}%** dan prediksi keberhasilan **{$confidence}%**. " .
               "Rekomendasi ini didasarkan pada analisis mendalam terhadap profil psikologis Anda, " .
               "kemampuan akademik, dan keselarasan dengan karakteristik jurusan.";
    }
    
    private function generateWhyThisMajor(
        PsychometricProfile $profile,
        RecommendationResult $recommendation,
        $ahpWeights
    ): string {
        // Find dominant criterion
        $dominantCriterion = collect($ahpWeights->weights)->sortDesc()->first();
        $dominantName = collect($ahpWeights->weights)->sortDesc()->keys()->first();
        $dominantWeight = round($dominantCriterion * 100, 0);
        
        // Get major's score on dominant criterion
        $majorScore = $recommendation->criteria_breakdown[$dominantName] ?? 0;
        
        $narrative = "**Mengapa {$recommendation->major->name} cocok untuk Anda?**\n\n";
        
        $narrative .= "1. **Keselarasan Prioritas Utama**\n";
        $narrative .= "   Prioritas utama Anda adalah **{$dominantName}** (bobot AHP: {$dominantWeight}%). ";
        $narrative .= "Jurusan {$recommendation->major->name} memiliki skor {$majorScore}/100 pada kriteria ini, ";
        $narrative .= "yang sangat sejalan dengan ekspektasi Anda.\n\n";
        
        // Identify top 3 strengths
        $strengths = $this->identifyTopStrengths($profile, $recommendation);
        $narrative .= "2. **Kekuatan Profil Anda**\n";
        foreach ($strengths as $i => $strength) {
            $narrative .= "   " . ($i + 1) . ". {$strength['name']}: {$strength['score']}/100 ";
            $narrative .= "({$strength['interpretation']})\n";
        }
        $narrative .= "\n";
        
        // Compensation analysis
        $weaknesses = $this->identifyWeaknesses($profile, $recommendation);
        if (!empty($weaknesses)) {
            $narrative .= "3. **Kompensasi Kelemahan**\n";
            $narrative .= "   Meskipun Anda memiliki kelemahan minor pada beberapa aspek, ";
            $narrative .= "hal ini dikompensasi oleh kekuatan Anda di area lain:\n";

    private function identifyTopStrengths(
        PsychometricProfile $profile,
        RecommendationResult $recommendation
    ): array {
        $attributes = [
            'academic_readiness' => 'Kesiapan Akademik',
            'logic_ability' => 'Kemampuan Logika',
            'analytical_ability' => 'Kemampuan Analitis',
            'field_interest' => 'Minat Bidang',
            'consistency' => 'Konsistensi',
            'grit_score' => 'Daya Tahan (Grit)'
        ];
        
        $scores = [];
        foreach ($attributes as $key => $name) {
            $score = $profile->$key;
            $threshold = $recommendation->major->{$key . '_threshold'} ?? 70;
            $gap = $score - $threshold;
            
            if ($gap >= 0) {
                $scores[] = [
                    'name' => $name,
                    'score' => $score,
                    'gap' => $gap,
                    'interpretation' => $this->interpretScore($score)
                ];
            }
        }
        
        // Sort by gap (highest first) and take top 3
        usort($scores, fn($a, $b) => $b['gap'] <=> $a['gap']);
        return array_slice($scores, 0, 3);
    }
    
    private function interpretScore(float $score): string
    {
        if ($score >= 95) return 'Luar Biasa';
        if ($score >= 90) return 'Sangat Tinggi';
        if ($score >= 85) return 'Tinggi';
        if ($score >= 80) return 'Baik';
        if ($score >= 75) return 'Cukup Baik';
        if ($score >= 70) return 'Memadai';
        return 'Perlu Peningkatan';
    }
```
